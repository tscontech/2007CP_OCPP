Example:

ret = ugCheckFilesCrc("B:", "B:/ite_crc.dat");
if (ret)
return ret; // check crc fail
ret = ugSetFileCrc("/font/DroidSans.ttf", "B:", "B:/ite_crc.dat");
if (ret)
return ret; // set crc fail
